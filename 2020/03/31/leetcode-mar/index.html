DFS

class Solution {
public:
    vector<vector<int>> res;   
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) 
    {
        addVector(root,0);      //调用递归函数
        return res;
    }
  
    void addVector(TreeNode* root,int level)
    {
        if(root == NULL)    return;
        if(res.size()==level)       res.resize(level+1);    //level表示层数，也对应二维数组的第一层索引，
        if(level %2==0)                                     
            res[level].push_back(root->val);                //如果为偶数行则顺序加入，如果为奇数行则将数字每次插入到最前面
        else
            res[level].insert(res[level].begin(),root->val);

        addVector(root->left,level+1);
        addVector(root->right,level+1);          
    }
};

class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> res;
        
        if (root == NULL)   return {};  
        queue<TreeNode*> q;     
        q.push(root);  

        int level=0;
        while (!q.empty()) 
        {  
            vector<int>temp;                //存放每一层的元素值
            int count=q.size();             //队列大小表示当前层数的元素个数
            while(count--)                  //count--逐个对该层元素进行处理
            {
            TreeNode *t=q.front();             q.pop();    
            if(level%2 ==0) 
            temp.push_back(t->val);
            else
            temp.insert(temp.begin(),t->val);
            if(t->left)     q.push(t->left);
            if(t->right)    q.push(t->right);
            }
            level++;
            res.push_back(temp);           //将当层元素的vector加入res中
        }
        return res;
        }
};

class Solution {
public:
    int majorityElement(vector<int>& nums) {
    int res=0;
    for(int i=0;i<32;i++)
    {
        int ones=0;
        for(int n:nums)
            ones += (n >> i) & 1;                        //位运算法统计每个位置上1出现的次数，每次出现则ones+1
        res += (ones > nums.size()/2) << i;              //如果1出现次数大于2分之1数组长，1即为这个位置的目标数字
    }
    return res;
}
};
class Solution {
    public int findDuplicate(int[] nums) {
        int slow = 0;
        int fast = 0;
        slow = nums[slow];
        fast = nums[nums[fast]];
        while(slow != fast){
            slow = nums[slow];
            fast = nums[nums[fast]];
        }
        int pre1 = 0;
        int pre2 = slow;
        while(pre1 != pre2){
            pre1 = nums[pre1];
            pre2 = nums[pre2];
        }
        return pre1;
    }
}
class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
        int len = nums.size();
        vector<int> res;
        if(len==0) return res;
        int count1=0 ,count2=0;
        int candidate1= 0 ,candidate2=0;

        for(auto x:nums)
        {//第一轮扫描确定candidate
            if(candidate1 ==x) count1++;
            else if(candidate2 == x) count2++;
            else if(count1==0)
            {
                candidate1 =x;
                count1=1;
            } 
            else if(count2==0)
            {
                candidate2 =x;
                count2=1;
            }
            else
            {
                count1--;
                count2--;
            }
        }
        cout<<candidate1<<". "<<candidate2;
        count1=0;
        count2=0;
        
        for(auto x : nums)
        {
            if(candidate1 ==x) count1++;
            else if(candidate2 ==x) count2++;
        }
        if(count1 > len/3) res.push_back(candidate1);
        if(count2 > len/3) res.push_back(candidate2);
        return res;

    }
};
